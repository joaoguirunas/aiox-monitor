- [ ] **Task 1** вҖ” Criar `src/hooks/useEvents.ts` (AC: 2, 4)
  ```typescript
  // Hook para buscar eventos com filtros
  export function useEvents(filters: EventFilters) {
    const [events, setEvents] = useState<Event[]>([]);
    const [loading, setLoading] = useState(true);
    const [total, setTotal] = useState(0);

    useEffect(() => {
      setLoading(true);
      fetch(`/api/events?${buildQueryString(filters)}`)
        .then(r => r.json())
        .then(data => {
          setEvents(data.events);
          setTotal(data.total);
          setLoading(false);
        });
    }, [filters]);

    return { events, loading, total };
  }
  ```
